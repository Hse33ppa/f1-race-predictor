    <script>

        function fetchPredictions() {

            const container = document.getElementById('prediction-container');

            container.innerHTML = '<div class="loading">Loading predictions...</div>';

            

            fetch('/predict')

                .then(response => response.json())

                .then(data => {

                    container.innerHTML = '';

                    data.forEach(prediction => {

                        const div = document.createElement('div');

                        div.className = 'driver-prediction';

                        div.innerHTML = `

                            <span class="driver-name">${prediction.driver}</span>

                            <span class="probability">${prediction.probability} chance to win</span>

                        `;

                        container.appendChild(div);

                    });

                })

                .catch(error => {

                    container.innerHTML = `<div class="loading">Error: ${error.message}</div>`;

                });

        }

    </script>
